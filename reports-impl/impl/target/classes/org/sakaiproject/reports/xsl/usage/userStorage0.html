<xsl:template name="chooseSize">
  <xsl:param name="num"/>
  <xsl:choose>
    <xsl:when test="($num div 1048576) >= 1">
      <xsl:value-of select="format-number($num div 1048576,'###,###.#')"/>GB</xsl:when>
    <xsl:when test="($num div 1024) >= 1">
      <xsl:value-of select="format-number($num div 1024,'###,###.#')"/>MB</xsl:when>
    <xsl:otherwise>
      <xsl:value-of select="format-number($num,'###,##0.0')"/>KB</xsl:otherwise>
  </xsl:choose>
</xsl:template>
